import { BrowserRouter as Router, Route, Routes } from "react-router-dom";
import Event from "./Event";
import EditEvent from "./EditEvent";

function App() {
  return (
    <Router>
      <Routes>
        <Route path="/" element={<Event />} />
        <Route path="/edit-event/:id" element={<EditEvent />} />
      </Routes>
    </Router>
  );
}

export default App;
